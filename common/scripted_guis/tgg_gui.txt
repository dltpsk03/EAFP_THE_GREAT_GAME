renew_gbr_influence_list = {
	effect = {
		clear_global_variable_list = gbr_influence_list
		every_country = {
			limit = {
				OR = {
					has_diplomatic_pact = {
							who = c:GBR
							type = alliance
					}
					has_diplomatic_pact = {
							who = c:GBR
							type = personal_union
					}
					has_diplomatic_pact = {
							who = c:GBR
							type = dominion
					}
					has_diplomatic_pact = {
							who = c:GBR
							type = customs_union
					}
					has_diplomatic_pact = {
							who = c:GBR
							type = puppet
					}
					has_diplomatic_pact = {
							who = c:GBR
							type = vassal
					}
					has_diplomatic_pact = {
							who = c:GBR
							type = protectorate
					}
				}
			}
			add_to_global_variable_list = {
				name = gbr_influence_list
				target = THIS
			}
		}
	}
}
