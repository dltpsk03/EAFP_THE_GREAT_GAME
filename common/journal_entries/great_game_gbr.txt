je_tgg_central_asia = {
	icon = "gfx/interface/icons/event_icons/event_military.dds"

	is_shown_when_inactive = {
		c:GBR = root
	}

	immediate = {
		set_global_variable = {
			name = calc_influence_gbr
			value = 0
		}
		add_modifier = {
			name = tgg_bonus
		}
	}

	scripted_button = tgg_central_asia


	possible = {
		c:GBR = root
		game_date > 1836.7.31
	}

	complete = {
		custom_tooltip = {
			text = great_game_win_tt
			has_global_variable = uzbek_win_gbr
			has_global_variable = persia_win_gbr
			has_global_variable = afghanistan_win_gbr
			has_global_variable = kyrgyz_win_gbr
		}
	}

	on_complete = {
		c:RUS = {
			trigger_event = {
				id = tgg_main.1
			}
		}
	}

	fail = {
		custom_tooltip = {
			text = great_game_lose_tt
			has_global_variable = uzbek_win_rus
			has_global_variable = persia_win_rus
			has_global_variable = afghanistan_win_rus
			has_global_variable = kyrgyz_win_rus
		}
	}
	
	on_fail = {
		trigger_event = {
			id = tgg_main.2
		}
	}

	invalid = {
		custom_tooltip = {
			text = great_game_end_tt
			has_global_variable = anglo_russo_treaty_var
		}
	}

	on_invalid = {
		trigger_event = {
			id = tgg_main.3
		}
	}

	weight = 9000

    should_be_pinned_by_default = yes
}

tgg_persia_coup = {
	icon = "gfx/interface/icons/event_icons/event_military.dds"

	immediate = {
		set_global_variable = {
			name = tgg_operation_gbr
			value = 0
		}
		calc_influence_gbr_effect = yes
	}

	on_monthly_pulse = {
		effect = {
			change_global_variable = {
				name = tgg_operation_gbr
				add = 1
			}
			switch = {
				trigger = global_var:tgg_operation_gbr
				1 = { 
					c:RUS = {
						trigger_event = tgg_main.54
					} 
				}
				10 = { 
					c:RUS = {
						trigger_event = tgg_main.55
					} 
				}
				20 = { 
					c:RUS = {
						trigger_event = tgg_main.56
					} 
				}
				fallback = { always = no }
			}
		}
	}

	complete = {
		scope:journal_entry = { is_goal_complete = yes }
	}

	on_complete = {
		c:RUS = {
			trigger_event = {
				id = tgg_main.57
			}
		}
		remove_global_variable = tgg_operation_gbr
		remove_global_variable = tgg_central_asia_coup
		remove_global_variable = tgg_event_on_central_asia
		calc_bureaucracy_rus_effect = yes
		calc_authority_rus_effect = yes
		calc_influence_rus_effect = yes
		
	}

	fail = {
		OR = {
			has_global_variable = tgg_operation_russia_win
			has_global_variable = tgg_operation_russia_intervention
			has_global_variable = tgg_operation_russia_threat
		}
	}

	on_fail = {
		if = {
			limit = { has_global_variable = tgg_operation_russia_win }
			c:RUS = {
				trigger_event = {
					id = tgg_main.58
				}
			}
		}
		else_if = {
			limit = { has_global_variable = tgg_operation_russia_intervention }
			c:RUS = {
				trigger_event = {
					id = tgg_main.59
				}
			}
		}
		else_if = {
			limit = { has_global_variable = tgg_operation_russia_threat }
			c:RUS = {
				trigger_event = {
					id = tgg_main.60
				}
			}
		}
		great_game_reset = yes
		calc_bureaucracy_rus_effect = yes
		calc_authority_rus_effect = yes
		calc_influence_rus_effect = yes
	}


	goal_add_value = {
		value = 30
	}

	current_value = {
		val0e = global_var:tgg_operation_gbr
	}

	weight = 9000

    progressbar = yes

	should_be_pinned_by_default = yes
}
